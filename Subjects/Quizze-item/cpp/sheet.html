1. Create a Calculator using Switch Statement

C++ 
#include <iostream>
using namespace std;

int main() {
    char oper;
    float num1, num2;
    cout << "Enter an operator (+, -, *, /): ";
    cin >> oper;
    cout << "Enter two numbers: " << endl;
    cin >> num1 >> num2;
    
    switch (oper) {
        case '+':
            cout << num1 << "+" << num2 << "=" << num1 + num2;
            break;
        case '-':
            cout << num1 << "-" << num2 << "=" << num1 - num2;
            break;
        case '*':
            cout << num1 << "*" << num2 << "=" << num1 * num2;
            break;
        case '/':
            cout << num1 << "/" << num2 << "=" << num1 / num2;
            break;
        default:
            cout << "Error! The operator is not correct";
            break;
    }
    return 0;
}

2. Add All Numbers from 1 to a Given Number

C++ 
#include <iostream>
using namespace std;

int Add_1_to_given_number(int n) {
    int total = 0;
    for (int x = 1; x <= n; x++) {
        total += x;
    }
    return total;
}

int main() {
    cout << "\nAdd 1 to 4: " << Add_1_to_given_number(4);
    cout << "\nAdd 1 to 100: " << Add_1_to_given_number(100);
    return 0;
}

4. Check Whether a Number is Prime or Not

C++ 
#include <iostream>
using namespace std;

int main() {
    int num1, ctr = 0;
    cout << "\n\n Check whether a number is prime or not:\n";
    cout << "-------------------------------------\n";
    cout << " Input a number to check prime or not: ";
    cin >> num1;
    
    for (int a = 1; a <= num1; a++) {
        if (num1 % a == 0) {
            ctr++;
        }
    }
    
    if (ctr == 2) {
        cout << " The entered number is a prime number. \n";
    } else {
        cout << " The number you entered is not a prime number. \n";
    }
    return 0;
}

7. Print All Even and Odd Numbers from 1 to N

C++ 
#include <iostream>
using namespace std;

void printEvenNumbers(int N) {
    cout << "Even: ";
    for (int i = 1; i <= 2 * N; i++) {
        if (i % 2 == 0)
            cout << i << " ";
    }
}

void printOddNumbers(int N) {
    cout << "\nOdd: ";
    for (int i = 1; i <= 2 * N; i++) {
        if (i % 2 != 0)
            cout << i << " ";
    }
}

int main() {
    int N = 5;
    printEvenNumbers(N);
    printOddNumbers(N);
    return 0;
}

9. Find Factorial of a Given Number N

C++ 
#include <iostream>
using namespace std;

int main() {
    int n;
    int factorial = 1;
    cout << "Enter a positive integer: ";
    cin >> n;
    
    if (n < 0)
        cout << "Error! Factorial of a negative number doesn't exist.";
    else {
        for(int i = 1; i <= n; ++i) {
            factorial *= i;
        }
        cout << "Factorial of " << n << " = " << factorial;
    }
    return 0;
}
